    def test_set_all_pins_off(self):
        """
        Testing set_all_pins function.
        Turn off all pins.
        :return:
        """
        val = (0, 0, 0, 0)
        set_all_pins(val)
        for p in PINS:
            p.value = MagicMock()
            p.value.return_value = 0
